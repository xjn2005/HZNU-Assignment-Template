\begin{problem}
Let $X$ be a random variable. Compute expectation and variance.
\end{problem}

\begin{solution}
\[
\E[X] = \sum_x x\,p(x), \qquad
\Var(X) = \E[X^2] - (\E[X])^2
\]
\end{solution}

% ---------- Problem 3 ----------
\begin{problem}
Implement quicksort.
\end{problem}

\begin{solution}
\begin{lstlisting}
def quicksort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[0]
    left = [x for x in arr[1:] if x < pivot]
    right = [x for x in arr[1:] if x >= pivot]
    return quicksort(left) + [pivot] + quicksort(right)
\end{lstlisting}
\end{solution}
